const int N=110;
int n,m;
int h[N], e[N], ne[N], idx;
int v[N], w[N], f[N][N];

void add(int a, int b) // add node(b) to head(a)
{
    e[idx]=b, ne[idx]=h[a], h[a]=idx++;
}

void dfs(int u)
{
    for(int i=h[u]; i!=-1; i=ne[i])
    {
        int son=e[i];
        dfs(son); // Êõ¥Êñ∞Â≠êËäÇÁÇπ
        for(int j=m-v[u]; j>=0; j--)
            for(int k=0; k<=j; k++)
                f[u][j]=max(f[u][j], f[u][j-k]+f[son][k]);
    }
    for(int i=m; i>=v[u]; i--) // ÂÆπÈáèÂ§ß‰∫éÁ≠â‰∫éÊ†πËäÇÁÇπÈúÄË¶ÅÂä†‰∏äÊ†πËäÇÁÇπÁöÑ‰ª∑ÂÄº
        f[u][i]=f[u][i-v[u]]+w[u];

        f[u][i]=0;

{

    int root;
    for(int i=1; i<=n; i++)
    {
        int p;

        if(p==-1) root=i;
        else add(p, i);
    }

    dfs(root);

}
